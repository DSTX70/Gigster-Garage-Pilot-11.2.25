set -euo pipefail

ensure_repo() {
  git rev-parse --is-inside-work-tree >/dev/null 2>&1 || { echo "Run inside your repo."; exit 1; }
}
add_env_hint() {
  grep -q "$1" .env.example 2>/dev/null || echo "$1" >> .env.example
}
