### 1) Add a Pa11y runner script that forces no-sandbox

FILE: scripts/ci/run-pa11y.mjs

```js
import pa11y from "pa11y";

const url = process.argv[2];
if (!url) {
  console.error("Usage: node scripts/ci/run-pa11y.mjs <url>");
  process.exit(2);
}

try {
  const result = await pa11y(url, {
    standard: "WCAG2AA",
    timeout: 60000,
    wait: 1000,
    includeNotices: true,
    includeWarnings: true,
    chromeLaunchConfig: {
      args: ["--no-sandbox", "--disable-setuid-sandbox", "--disable-dev-shm-usage"],
    },
  });

  process.stdout.write(JSON.stringify(result, null, 2));

  // Big Audit v0: informational (don’t fail the whole workflow yet)
  process.exit(0);
} catch (err) {
  console.error("Pa11y execution failed:", err?.message || err);
  process.exit(1);
}
```

### 2) Update the workflow to (a) run Pa11y via the script and (b) always upload artifacts

FILE: .github/workflows/website-audit.yml

```yaml
name: Website Audit (Lighthouse + a11y)

on:
  workflow_dispatch:
  pull_request:
    branches: ["main"]
  push:
    branches: ["main"]

concurrency:
  group: website-audit-${{ github.ref }}
  cancel-in-progress: true

jobs:
  audit:
    runs-on: ubuntu-latest
    timeout-minutes: 20

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: "20"

      - name: Verify AUDIT_BASE_URL secret
        env:
          AUDIT_BASE_URL: ${{ secrets.AUDIT_BASE_URL }}
        run: |
          if [ -z "${AUDIT_BASE_URL}" ]; then
            echo "ERROR: Missing GitHub Actions secret AUDIT_BASE_URL"
            exit 1
          fi
          echo "AUDIT_BASE_URL is set."

      # Run Lighthouse as separate stages so .lighthouseci persists reliably
      - name: Lighthouse CI - collect
        env:
          AUDIT_BASE_URL: ${{ secrets.AUDIT_BASE_URL }}
        run: |
          echo "Collecting Lighthouse runs against: ${AUDIT_BASE_URL}"
          npx -y @lhci/cli collect \
            --config=.lighthouserc.json \
            --url="${AUDIT_BASE_URL}" \
            --numberOfRuns=3

      - name: Lighthouse CI - assert (warn-only thresholds live in config)
        run: |
          npx -y @lhci/cli assert --config=.lighthouserc.json

      - name: Lighthouse CI - upload (temporary public storage)
        run: |
          npx -y @lhci/cli upload --config=.lighthouserc.json --target=temporary-public-storage

      - name: Debug: list lighthouse output dir
        if: always()
        run: |
          echo "Listing .lighthouseci (if present):"
          ls -la .lighthouseci || true

      - name: Install Pa11y (direct runner)
        run: |
          npm i --no-save pa11y

      - name: Run Pa11y (WCAG2AA) — outputs JSON report
        env:
          AUDIT_BASE_URL: ${{ secrets.AUDIT_BASE_URL }}
        run: |
          echo "Running Pa11y (direct) against: ${AUDIT_BASE_URL}"
          node scripts/ci/run-pa11y.mjs "${AUDIT_BASE_URL}" > pa11y-report.json
          echo "Pa11y report saved to pa11y-report.json"

      - name: Upload Lighthouse artifacts
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: lighthouseci-artifacts
          path: .lighthouseci
          if-no-files-found: ignore

      - name: Upload Pa11y report
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: pa11y-report
          path: pa11y-report.json
          if-no-files-found: ignore
```

END_FILE
