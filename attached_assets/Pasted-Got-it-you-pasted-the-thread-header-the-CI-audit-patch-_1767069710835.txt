      - name: Lighthouse CI (uses .lighthouserc.json for assertions)

          AUDIT_BASE_URL: ${{ secrets.AUDIT_BASE_URL }}

          echo "Running Lighthouse CI against: ${AUDIT_BASE_URL}"

            --config=.lighthouserc.json \
