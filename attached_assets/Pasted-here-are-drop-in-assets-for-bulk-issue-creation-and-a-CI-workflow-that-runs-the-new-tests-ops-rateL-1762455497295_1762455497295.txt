Importer script (uses gh CLI):

/ops/github/import_issues.sh

#!/usr/bin/env bash
set -euo pipefail
FILE="${1:-ops/github/issues.json}"

which gh >/dev/null || { echo "Please install GitHub CLI: https://cli.github.com/"; exit 1; }

jq -c '.[]' "$FILE" | while read -r issue; do
  TITLE=$(echo "$issue" | jq -r '.title')
  BODY=$(echo "$issue"  | jq -r '.body')
  LABELS=$(echo "$issue"| jq -r '.labels | join(",")')
  ASSIGNEES=$(echo "$issue" | jq -r '.assignees | join(",")')
  gh issue create --title "$TITLE" --body "$BODY" \
    ${LABELS:+--label "$LABELS"} \
    ${ASSIGNEES:+--assignee "$ASSIGNEES"}
done


Run:

chmod +x ops/github/import_issues.sh
./ops/github/import_issues.sh ops/github/issues.json

2) GitHub Actions — CI for ops.rateLimits.*, schema diffs, social queue

/.github/workflows/ci.yml

name: CI

on:
  push:
    branches: [ "**" ]
  pull_request:

jobs:
  test:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_DB: appdb
          POSTGRES_USER: app
          POSTGRES_PASSWORD: secret
        ports:
          - 5432:5432
        options: >-
          --health-cmd="pg_isready -U app -d appdb"
          --health-interval=10s
          --health-timeout=5s
          --health-retries=5

    env:
      DATABASE_URL: postgresql://app:secret@localhost:5432/appdb
      NODE_ENV: test
      SOCIAL_WORKER_POLL_MS: 1000

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: "20"
          cache: "npm"

      - name: Install
        run: npm ci

      - name: Wait for Postgres
        run: |
          for i in {1..20}; do
            pg_isready -h localhost -p 5432 -U app && break
            sleep 1
          done

      - name: Run DB migrations
        run: |
          # Include all SQL migrations your repo has (add more as needed)
          find server/db/migrations -maxdepth 1 -name "*.sql" -print0 | sort -z | xargs -0 -I{} psql "$DATABASE_URL" -f {}

      - name: Lint
        run: npm run lint --if-present

      - name: Build
        run: npm run build --if-present

      - name: Tests — rate-limits + schema + social queue
        run: |
          npx jest \
            "server/tests/ops.rateLimits.*.spec.ts" \
            "server/tests/schema.*.spec.ts" \
            "server/tests/social.*.spec.ts" \
            --runInBand --detectOpenHandles --forceExit

      - name: Upload coverage (optional)
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: coverage
          path: coverage


Notes:

The workflow brings up Postgres and runs all SQL migrations in server/db/migrations/.

It runs three test globs:

ops.rateLimits.*.spec.ts (your dashboard/CSV/window tests)

schema.*.spec.ts (response vs Zod)

social.*.spec.ts (social queue tests)

Adjust paths if your repo differs.

3) Optional: tiny parser for .gh-issues.md → create issues

/ops/github/create_from_md.sh

#!/usr/bin/env bash
set -euo pipefail
FILE="${1:-.gh-issues.md}"

# Requires: yq, gh
which yq >/dev/null || { echo "Install yq: https://mikefarah.gitbook.io/yq/"; exit 1; }
which gh >/dev/null || { echo "Install gh: https://cli.github.com/"; exit 1; }

awk -v RS="---\n" 'NR>1{print >("issue_" NR ".md")}' "$FILE"

for f in issue_*.md; do
  title=$(yq '.title' "$f")
  labels=$(yq -o=csv '.labels' "$f" | tr -d '"')
  assignees=$(yq -o=csv '.assignees' "$f" | tr -d '"')
  body=$(awk 'BEGIN{p=0} /^---$/{p=1;next} {if(p)print}' "$f")
  gh issue create --title "$title" --body "$body" \
    ${labels:+--label "$labels"} \
    ${assignees:+--assignee "$assignees"}
  rm "$f"
done
