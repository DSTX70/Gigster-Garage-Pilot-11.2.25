## 2) NEW FILE: `client/src/components/dashboard/ActionCard.tsx`

Create the file and paste:

```tsx
import * as React from "react";
import { Link } from "wouter";
import { Card, CardContent } from "@/components/ui/card";
import { cn } from "@/lib/utils";

type ActionCardProps = {
  href: string;
  title: string;
  description: string;
  icon: React.ReactNode;
  borderClassName?: string;   // e.g. "border-l-blue-500"
  iconWrapClassName?: string; // e.g. "bg-blue-100"
};

export function ActionCard({
  href,
  title,
  description,
  icon,
  borderClassName,
  iconWrapClassName,
}: ActionCardProps) {
  return (
    <Link href={href}>
      <Card
        className={cn(
          "hover:shadow-lg transition-shadow cursor-pointer",
          borderClassName ? `border-l-4 ${borderClassName}` : ""
        )}
      >
        <CardContent className="p-4 sm:p-6">
          <div className="flex items-center gap-3 sm:gap-4">
            <div className={cn("p-2 sm:p-3 rounded-lg", iconWrapClassName ?? "bg-gray-100")}>
              {icon}
            </div>
            <div>
              <h3 className="text-sm sm:text-base font-semibold text-gray-900">{title}</h3>
              <p className="text-xs sm:text-sm text-gray-600">{description}</p>
            </div>
          </div>
        </CardContent>
      </Card>
    </Link>
  );
}
```
